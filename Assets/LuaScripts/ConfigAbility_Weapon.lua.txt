        name = "ShenHaiFangKe",

            IceDmgPct = 0.25,
        },

                self:attachAbility("ShenHaiFangKe_2")
            end
        end,

    {
        name = "ShenHaiFangKe_1",

        end

        duration = 15,
        attrs = {
            CritRate = 0.1
        }
    },
    {
        name = "LiuHuangShi",

                if not self:hasTimer("LiuHuangShi_Timer") then
                    self:setTimer("LiuHuangShi_Timer", 0.5)
                    self:attachAbility("LiuHuangShi_1")
                end
            end
        end
    },

        name = "LiuHuangShi_1",
